<script>

function showPage(id){
 document.querySelectorAll('.page').forEach(p=>p.classList.remove('active'))
 document.getElementById(id).classList.add('active')
}

function mod(v){return Math.floor((v-10)/2)}

function log(t){
let l=document.getElementById("log")
let p=document.createElement("div")
p.innerText=t
l.prepend(p)
}

function getMod(attr){return mod(parseInt(document.getElementById(attr).value||0))}

function updateStats(){
let attrs=["for","des","con","int","sab","car"]
attrs.forEach(a=>{
let m=getMod(a)
let txt=m>=0?"+"+m:m

document.getElementById(a+"_mod").innerText=txt
if(a=="des")document.getElementById("init").innerText=txt
})
updateHP()
}

function updateHP(){
let hp=document.getElementById("hp").value
let max=document.getElementById("hpmax").value
let p=(hp/max)*100
if(p<0)p=0
if(p>100)p=100

document.getElementById("hpinner").style.width=p+"%"
}

function rollInitiative(){
let base=Math.floor(Math.random()*20)+1
let total=base+getMod("des")
log("Iniciativa: "+total)
}

/* ATAQUES */

let attacks=[]

function addAttack(){
let name=document.getElementById("attack_name").value
let attr=document.getElementById("attack_attr").value
let dice=document.getElementById("attack_dice").value

attacks.push({name,attr,dice})
renderAttacks()
}

function rollAttack(attr,name){
let atk=Math.floor(Math.random()*20)+1+getMod(attr)
log(name+" Ataque: "+atk)
}

function rollDamage(dice,name){
let dmg=rollDice(dice)
log(name+" Dano: "+dmg)
}

function rollBoth(attr,dice,name){
rollAttack(attr,name)
rollDamage(dice,name)
}

function renderAttacks(){
let div=document.getElementById("attacks")
div.innerHTML=""

attacks.forEach((a,i)=>{
let el=document.createElement("div")
el.className="attackItem"

el.innerHTML=`
<b>${a.name}</b><br>
Dano: ${a.dice}<br>
<button onclick="rollAttack('${a.attr}','${a.name}')">Ataque</button>
<button onclick="rollDamage('${a.dice}','${a.name}')">Dano</button>
<button onclick="rollBoth('${a.attr}','${a.dice}','${a.name}')">Ambos</button>
`

div.appendChild(el)

})
}

/* ROLAGEM DE DADOS */

function rollDice(expr){
expr=expr.replace(/\s/g,'')

let bonus=0

if(expr.includes('+')){
let parts=expr.split('+')
expr=parts[0]
bonus=parseInt(parts[1])
}

let p=expr.split('d')
let q=parseInt(p[0])
let d=parseInt(p[1])

let total=0

for(let i=0;i<q;i++) total+=Math.floor(Math.random()*d)+1

return total+bonus
}

/* INVENTARIO */

let inventario=JSON.parse(localStorage.getItem("inventario")) || []

function saveInv(){
 localStorage.setItem("inventario",JSON.stringify(inventario))
}

function rarityClass(r){
if(r=="Uncommon")return"itemUncommon"
if(r=="Rare")return"itemRare"
if(r=="Epic")return"itemEpic"
if(r=="Legendary")return"itemLegendary"
return"itemCommon"
}

function addItem(){
let nome=document.getElementById("item_nome").value
let desc=document.getElementById("item_desc").value
let rarity=document.getElementById("item_rarity").value
let fileInput=document.getElementById("item_img")
let file=fileInput.files[0]

if(!nome)return

if(file){
 const reader=new FileReader()
 reader.onload=function(e){
  inventario.push({nome,desc,rarity,img:e.target.result})
  saveInv()
  renderInv()
 }
 reader.readAsDataURL(file)
}else{
 inventario.push({nome,desc,rarity,img:null})
 saveInv()
 renderInv()
}

fileInput.value=""
}

function removeItem(i){
inventario.splice(i,1)
saveInv()
renderInv()
}

function renderInv(){
let div=document.getElementById("inventario")
div.innerHTML=""

document.getElementById("itemCount").innerText=inventario.length

inventario.forEach((item,i)=>{

let slot=document.createElement("div")
slot.className="itemSlot "+rarityClass(item.rarity)

let image=item.img || "https://via.placeholder.com/64"

slot.innerHTML=`
<img src="${image}" style="width:60px;height:60px;object-fit:cover;border-radius:6px">
<div class="itemTooltip"><b>${item.nome}</b><br>${item.rarity}<br>${item.desc}</div>
<button onclick="removeItem(${i})" style="position:absolute;bottom:2px;right:2px;font-size:10px">X</button>
`

div.appendChild(slot)

})
}

renderInv()

/* MAGIAS */

let spellsByLevel={}
const spellLevelsDiv=document.getElementById("spellLevels")
const spellLevelSelect=document.getElementById("spell_level")

for(let i=0;i<=9;i++){

spellsByLevel[i]={slots:0,spells:[]}

let option=document.createElement("option")
option.value=i
option.text=i+"º círculo"
spellLevelSelect.appendChild(option)

let div=document.createElement("div")
div.className="spellLevel"

div.innerHTML=`
<b>${i}º círculo</b>
<div>Slots: <input type="number" value="0" onchange="setSlots(${i},this.value)" style="width:60px"></div>
<div id="spell_list_${i}"></div>
`

spellLevelsDiv.appendChild(div)
}

function setSlots(level,value){
spellsByLevel[level].slots=parseInt(value)
}

function addSpell(){
let name=document.getElementById("spell_name").value
let level=document.getElementById("spell_level").value
let dice=document.getElementById("spell_dice").value
let attr=document.getElementById("spell_attr").value

spellsByLevel[level].spells.push({name,dice,attr})

renderSpells(level)
}

function spellAttack(attr,name){
let atk=Math.floor(Math.random()*20)+1+getMod(attr)
log(name+" Ataque mágico: "+atk)
}

function spellDamage(dice,name){
let dmg=rollDice(dice)
log(name+" Dano mágico: "+dmg)
}

function spellBoth(attr,dice,name,level){
let data=spellsByLevel[level]

if(data.slots<=0){
alert("Sem slots desse círculo")
return
}

data.slots--

spellAttack(attr,name)
spellDamage(dice,name)

renderSpells(level)
}

function renderSpells(level){
let list=document.getElementById("spell_list_"+level)
let data=spellsByLevel[level]

list.innerHTML="Slots restantes: "+data.slots+"<br>"

data.spells.forEach((s,i)=>{

let el=document.createElement("div")
el.innerHTML=`
${s.name} (${s.dice})
<button onclick="spellAttack('${s.attr}','${s.name}')">Ataque</button>
<button onclick="spellDamage('${s.dice}','${s.name}')">Dano</button>
<button onclick="spellBoth('${s.attr}','${s.dice}','${s.name}',${level})">Ambos</button>
`

list.appendChild(el)

})
}
